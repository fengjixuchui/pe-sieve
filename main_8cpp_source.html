67size_t print_params_block(std::string block_name, std::map<std::string, void(*)(int)> params_block, const std::string &filter)
68{
69 const int hdr_color = HEADER_COLOR;
70 const int param_color = HILIGHTED_COLOR;
71 const int separator_color = SEPARATOR_COLOR;
72
73 const bool has_filter = filter.length() > 0 ? true : false;
74 bool has_any = false;
75
76 std::map<std::string, void(*)(int)>::iterator itr;
77 for (itr = params_block.begin(); itr != params_block.end(); ++itr) {
78 const std::string &param = itr->first;
79 if (has_filter) {
80 stringsim_type sim_type = is_string_similar(param, filter);
81 if (sim_type != SIM_NONE) has_any = true;
82 }
83 else {
84 has_any = true;
85 }
86 }
87 if (!has_any) return 0;
88
89 int p_color = param_color;
90 if (block_name.length()) {
91 print_in_color(separator_color, "\n---" + block_name + "---\n");
92 }
93 size_t counter = 0;
94 for (itr = params_block.begin(); itr != params_block.end(); ++itr) {
95 const std::string &param = itr->first;
96 if (filter.length() > 0) {
97 const stringsim_type sim_type = is_string_similar(param, filter);
98 p_color = (sim_type != SIM_NONE) ? ERROR_COLOR : param_color;
99 if (sim_type == SIM_NONE) continue;
100 }
101 void(*info)(int) = itr->second;
102 if (!info) continue;
103 info(p_color);
104 counter++;
105 }
106 if (has_filter) {
107 print_in_color(INACTIVE_COLOR, "\n[...]\n");
108 }
109 return counter;
110}

112size_t copyToCStr(char *buf, size_t buf_max, const std::string &value)
113{
114 size_t len = value.length() + 1;
115 if (len > buf_max) len = buf_max;
116
117 memset(buf, 0, buf_max);
118 memcpy(buf, value.c_str(), len);
119 buf[len] = '\0';
120 return len;
121}
