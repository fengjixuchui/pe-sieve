732 PeArtefacts* pesieve::ArtefactScanner::findArtefacts(MemPageData &memPage, size_t start_offset)
733 {
734  if (!memPage.load()) {
735  return nullptr;
736  }
737 
738  ArtefactsMapping bestMapping(memPage, this->isProcess64bit);
739 
740  for (size_t min_offset = start_offset; min_offset < memPage.getLoadedSize(); min_offset++)
741  {
742  //std::cout << "Searching DOS header, min_offset: " << std::hex << min_offset << std::endl;
743 
744  ArtefactsMapping aMap(memPage, this->isProcess64bit);
745  //try to find the DOS header
746  if (findMzPe(aMap, min_offset)) {
747  const size_t dos_offset = calc_offset(memPage, aMap.dos_hdr);
748  min_offset = (dos_offset != INVALID_OFFSET) ? dos_offset : min_offset;
749 #ifdef _DEBUG
750  std::cout << std::hex << "Page: " << aMap.memPage.start_va << " Found DOS Header at: " << dos_offset << "\n";
751 #endif
752  }
753  else {
754 #ifdef _DEBUG
755  std::cout << std::hex << "Page: " << aMap.memPage.start_va << " Searching NT Header at: " << min_offset << "\n";
756 #endif
757  IMAGE_FILE_HEADER *nt_hdr = findNtFileHdr(aMap.memPage, min_offset, memPage.getLoadedSize());
758  setNtFileHdr(aMap, nt_hdr);
759  }
760 
761  //adjust constraints for further searches:
762  size_t max_section_search = memPage.getLoadedSize();
763  if (aMap.nt_file_hdr) {
764  const size_t nt_offset = calc_offset(memPage, aMap.nt_file_hdr);
765  if (nt_offset != INVALID_OFFSET && nt_offset > min_offset) {
766  min_offset = nt_offset;
767  }
768  //don't search sections in full module, only in the first mem page after the NT header:
769  max_section_search = (PAGE_SIZE < memPage.getLoadedSize()) ? PAGE_SIZE : memPage.getLoadedSize();
770  if (max_section_search + min_offset <= memPage.getLoadedSize()) {
771  max_section_search += min_offset; //move the search window
772  }
773  }
774 
775  if (!setSecHdr(aMap, aMap.sec_hdr)) {
776  //search sections by pattens:
777  if (max_section_search > min_offset) {
778  const size_t diff = max_section_search - min_offset;
779  IMAGE_SECTION_HEADER *sec_hdr = findSecByPatterns(memPage, diff, min_offset);
780  setSecHdr(aMap, sec_hdr);
781  }
782  }
783  if (aMap.sec_hdr) {
784  const size_t sec_offset = calc_offset(memPage, aMap.sec_hdr);
785  if (sec_offset != INVALID_OFFSET && sec_offset > min_offset) {
786  const size_t sections_area_size = aMap.sec_count * sizeof(IMAGE_SECTION_HEADER);
787  min_offset = (sec_offset + sections_area_size);
788 #ifdef _DEBUG
789  std::cout << "Setting minOffset to SecHdr end offset: " << std::hex << min_offset << "\n";
790 #endif
791  }
792 
793  if (!aMap.dos_hdr) {
794  const size_t start = (sec_offset > PAGE_SIZE) ? (sec_offset - PAGE_SIZE) : 0;
795  //std::cout << "Searching DOS header by patterns " << std::hex << start << "\n";
796  aMap.dos_hdr = findDosHdrByPatterns(aMap.memPage, start, sec_offset);
797  if (aMap.dos_hdr && !aMap.nt_file_hdr) {
798  IMAGE_NT_HEADERS32 *nt_ptr = (IMAGE_NT_HEADERS32*)((ULONG_PTR)aMap.dos_hdr + aMap.dos_hdr->e_lfanew);
799 #ifdef _DEBUG
800  const size_t nt_offset = calc_offset(memPage, nt_ptr);
801  std::cout << "Found PE offset: " << std::hex << aMap.dos_hdr->e_lfanew << " NT offset: " << nt_offset << "\n";
802 #endif
803  if (aMap.memPage.validatePtr(nt_ptr, sizeof(IMAGE_NT_HEADERS32))) {
804  setNtFileHdr(aMap, &nt_ptr->FileHeader);
805  }
806  }
807  }
808  }
809  if (!setSecHdr(aMap, aMap.sec_hdr)) {
810  aMap.sec_hdr = nullptr;
811  }
812  bestMapping = (bestMapping < aMap) ? aMap : bestMapping;
813 
814  //do not continue the search if no artefacts found:
815  if (!aMap.foundAny()) break;
816  }
817  //use the best found set of artefacts:
818  return generateArtefacts(bestMapping);
819 }
