268  const ULONGLONG addr = itr->second;
269 
270  ScannedModule *modExp = modulesInfo.findModuleContaining(addr);
271  ULONGLONG module_start = (modExp) ? modExp->getStart() : peconv::fetch_alloc_base(this->processHandle, (BYTE*)addr);
272  if (module_start == 0) {
